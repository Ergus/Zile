\reftable{Help}{
\kbd{C-h h}, \key{F1}           & Show the Zile manual          \\
\kbd{C-h C-r}, \key{F9}         & Go to next \textit{Mini Help} page \\
\kbd{C-h C-h}, \key{F10}        & Toggle \textit{Mini Help} window \\
\kbd{C-h F}                     & Show the FAQ                  \\
\kbd{C-h d}, \kbd{C-h f}        & Describe function             \\
\kbd{C-h k}                     & Describe key                  \\
\kbd{C-h v}                     & Describe variable             \\
\kbd{C-h l b}                   & List defined bindings         \\
\kbd{C-h l f}                   & List defined functions        \\
\kbd{C-h l r}                   & List defined registers        \\
\kbd{C-h l v}                   & List defined variables        \\
\kbd{C-h s}                     & Show configuration sample     \\
\kbd{C-h t}                     & Show tutorial                 \\
\kbd{C-h C-d}                   & Show Zile availability info   \\
}
\reftable{Cursor Motion}{
\kbd{C-v}, \key{PGDN}           & Scroll up                     \\
\kbd{M-v}, \key{PGUP}           & Scroll down                   \\
\kbd{C-a}, \key{HOME}           & Go to beginning of line       \\
\kbd{C-e}, \key{END}            & Go to end of line             \\
\kbd{C-f}, \key{RIGHT}          & Forward character             \\
\kbd{C-b}, \key{LEFT}           & Backward character            \\
\kbd{C-p}, \key{UP}             & Go to previous line           \\
\kbd{C-n}, \key{DOWN}           & Go to next line               \\
\kbd{M-<}                       & Go to beginning of buffer     \\
\kbd{M->}                       & Go to end of buffer           \\
\kbd{M-g}                       & Go to specified line          \\
}
\reftable{Buffers}{

\kbd{C-x i}                     & Insert file                   \\
\kbd{C-x C-s}, \key{F3}         & Save current buffer           \\
\kbd{C-x C-w}                   & Write to another file         \\
\kbd{C-x s}                     & Save modified buffers         \\
\kbd{C-x b}                     & Switch to another buffer      \\
\kbd{C-x C-b}                   & List buffers                  \\
\kbd{C-x k}                     & Kill current buffer           \\
